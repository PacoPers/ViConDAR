% Create .inp file from lidar measurements
%
% V.Pettas/F.Costa
% University of Stuttgart, Stuttgart Wind Energy (SWE) 2019

function CreateInpTurbsim(SaveName,TimeSeriesName,AnalysisTime,input,lidarMeas)

% Change the name of the time series in order to obtain the .wnd of the constrained WF:
% read the template for time series:

TurbsimFile=fileread(input.TurbSimInpTemplate);

% Look for the line I want to change
expr={ '[^\n]*UserFile[^\n]*','[^\n]*RandSeed1[^\n]*','[^\n]*NumGrid_Z[^\n]*','[^\n]*NumGrid_Y[^\n]*','[^\n]*TimeStep[^\n]*','[^\n]*AnalysisTime[^\n]*',...
    '[^\n]*unless[^\n]*','[^\n]*HubHt[^\n]*','[^\n]*GridHeight[^\n]*','[^\n]*GridWidth[^\n]*','[^\n]*IECstandard[^\n]*','[^\n]*IECturbc[^\n]*',...
    '[^\n]*WindProfileType[^\n]*','[^\n]*RefHt[^\n]*','[^\n]*total[^\n]*','[^\n]*PLExp[^\n]*','[^\n]*SCMod1[^\n]*'};

for ind_exp=1:length(expr)
    matches{ind_exp}= (regexp((TurbsimFile),expr{ind_exp},'match','once'));
    matches{ind_exp}=char(matches{ind_exp});
    exp{ind_exp} = strsplit(matches{1,ind_exp});
end

%Change name of time series:
expres= '"(.*?)"';% text between quotation marks
matchStr0 = regexp(matches{1},expres,'match','once');
NewTimeSeries=['"' TimeSeriesName '"']; %Name of the constrained Time series
NewTimeSeries = strrep(NewTimeSeries,'\','/');
NewTurbSimFile=regexprep(TurbsimFile,matchStr0,NewTimeSeries);

%Change the random seed:

matchStr1 = regexp(matches{2},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr1,num2str(input.RandSeed ));


%Change the GridZ:
matchStr2 = regexp(matches{3},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr2,num2str(lidarMeas.TS.fullWF.nGridZ),'once');

%Change the NumGrid_Y:
matchStr3 = regexp(matches{4},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr3,num2str(lidarMeas.TS.fullWF.nGridY),'once');

%Change the TimeStep:
matchStr4 = regexp(matches{5},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr4,num2str(lidarMeas.Pattern.timestep_pat_vec  ),'once');

%Change the AnalysisTime:
matchStr5 = regexp(matches{6},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr5,num2str(AnalysisTime(end)),'once');

%Change the UsableTime:
matchStr6 = regexp(matches{7},expres,'match','once');
if isnumeric(input.UsableTime)
    NewTurbSimFile=regexprep(NewTurbSimFile,matchStr6,num2str(input.UsableTime),'once');
else
    NewTurbSimFile=regexprep(NewTurbSimFile,matchStr6,(input.UsableTime),'once');
end

%Change the HubHt:
matchStr7 = regexp(matches{8},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr7,num2str(input.Zh ),'once');

%Change the GridHeight:

matchStr8 = regexp(matches{9},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr8,num2str((lidarMeas.TS.fullWF.nGridZ-1)*lidarMeas.TS.fullWF.dz),'once');

%Cahnge the GridWidth:

matchStr9 = regexp(matches{10},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr9,num2str((lidarMeas.TS.fullWF.nGridY-1)*lidarMeas.TS.fullWF.dy),'once');

%Cahnge the IECstandard:
matchStr10 = regexp(matches{11},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr10,input.IECstandard,'once');

%Cahnge the IECturbc:
matchStr11 = regexp(matches{12},expres,'match','once');
if isnumeric(lidarMeas.statistics.U.lidar.TI_mean)
    NewTurbSimFile=regexprep(NewTurbSimFile,matchStr11,['"',num2str(100*lidarMeas.statistics.U.lidar.TI_mean),'"'],'once');
else
    NewTurbSimFile=regexprep(NewTurbSimFile,matchStr11,(lidarMeas.statistics.U.lidar.TI_mean),'once');
end

%Cahnge the RefHt:
matchStr12 = regexp(matches{13},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr12,input.WindProfileType,'once');

%Cahnge the RefHt:
matchStr13 = regexp(matches{14},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr13,num2str(input.Zh),'once');

%Cahnge the URef:
matchStr14 = regexp(matches{15},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr14,num2str(lidarMeas.statistics.U.lidar.Mean_TS_LiDAR),'once');

%Cahnge the PLExp:
matchStr15 = regexp(matches{16},expres,'match','once');
if isnumeric (lidarMeas.Shear.lidar.Mean)
    if  lidarMeas.Shear.lidar.Mean<0
        lidarMeas.Shear.lidar.Mean=0;
    end
    lidarMeas.Shear.lidar.Mean=sprintf('%.3f',lidarMeas.Shear.lidar.Mean);
    NewTurbSimFile=regexprep(NewTurbSimFile,matchStr15,['"',num2str(lidarMeas.Shear.lidar.Mean),'"'],'once');
else
    NewTurbSimFile=regexprep(NewTurbSimFile,matchStr15,(lidarMeas.Shear.lidar.Mean),'once');
end
% Change SCMOD1:

matchStr16 = regexp(matches{17},expres,'match','once');
NewTurbSimFile=regexprep(NewTurbSimFile,matchStr16,input.SCMod1,'once');

%Write the file
fid = fopen(SaveName,'w');
fwrite(fid,NewTurbSimFile);
fclose(fid);


end
